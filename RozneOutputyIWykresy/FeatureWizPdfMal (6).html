def get_top_correlated_features(X_features, method_name, threshold=0.3, num_features=10):
    print()
    print()
    """
    Zwraca list num_features najbardziej skorelowanych cech na podstawie danej ramki danych X_features.

    Parameters:
    - X_features: pandas DataFrame, ramka danych zawierajca cechy
    - threshold: float, prg korelacji, powyej ktrego cechy s uznawane za mocno skorelowane (domylnie 0.8)
    - num_features: int, liczba cech do wyboru (domylnie 6)

    Returns:
    - top_correlated_features: list, lista zawierajca num_features najbardziej skorelowanych cech
    """
    correlation_matrix = np.corrcoef(X_features, rowvar=False)
    correlation_pairs = []

    # Tworzymy pary cech wraz z wartociami bezwzgldnymi korelacji
    for i in range(len(correlation_matrix.columns)):
        for j in range(i + 1, len(correlation_matrix.columns)):
            feature_pair = (correlation_matrix.columns[i], correlation_matrix.columns[j], abs(correlation_matrix.iloc[i, j]))
            correlation_pairs.append(feature_pair)

    # Sortujemy pary cech wedug wartoci bezwzgldnych korelacji w malejcej kolejnoci
    correlation_pairs.sort(key=lambda x: x[2], reverse=True)

    # Wybieramy num_features najbardziej skorelowanych cech
    top_correlated_features = []
    i = 0
    while len(top_correlated_features) < num_features and i < len(correlation_pairs):
        feature_pair = correlation_pairs[i]
        if feature_pair[0] not in top_correlated_features and feature_pair[1] not in top_correlated_features:
            top_correlated_features.append(feature_pair[0])
            top_correlated_features.append(feature_pair[1])
        i += 1

    # Wywietlamy najbardziej skorelowane cechy w danym formacie
    print(f"{num_features} najbardziej skorelowane cechy dla metody: {method_name}")
    for feature_pair in correlation_pairs[:num_features]:
        print(f"{feature_pair[0]} - {feature_pair[1]}: {feature_pair[2]:.5f}")

    return top_correlated_features[:num_features]

def plot_correlation_matrix(features, column_names):
    correlation_matrix = np.corrcoef(features, rowvar=False)
    # Utwrz wykres heatmapy
    plt.figure(figsize=(10, 10))  # Zmniejsz rozmiar wykresu
    heatmap = sns.heatmap(correlation_matrix, annot=True, cmap='coolwarm', fmt=".2f", linewidths=.5, cbar_kws={'label': 'Korelacja'})
    # Dostosuj kolor i rozmiar tekstu na osiach x, y i osi kolorw (cbar)
    heatmap.set_xticklabels(column_names, fontsize=9, rotation=45, ha='right')
    heatmap.set_yticklabels(column_names, fontsize=9)
    # Dostosuj kolor i rozmiar tekstu na osi kolorw (cbar)
    cbar = heatmap.collections[0].colorbar
    cbar.ax.tick_params(labelsize=12)
    # Zamiast cbar.set_ylabel, moemy skorzysta z tytuu osi kolorw
    cbar.set_label('Korelacja', fontsize=12)
    plt.title('Macierz Korelacji dla metody featurewiz ', fontsize=16)
    #plt.tight_layout()  # Dodaj tight_layout, aby poprawi ukad wykresu
    plt.show()
